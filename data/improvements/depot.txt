caravansary = {
	name = "Caravansary"
	slot = depot
	icon = trade_consulate
	wealth_cost = 200
	modifier = {
		depot_level = 2
	}
	conditions = {
		or = {
			has_pathway = trail
			has_pathway = road
			has_pathway = railroad
		}
	}
	free_on_start_conditions = {
		highest_production_capacity >= 1
		or = {
			not = {
				any_adjacent_site = {
					improvement = caravansary
				}
			}
			any_adjacent_site = {
				highest_production_capacity >= 1
				not = {
					improvement = caravansary
				}
			}
		}
	}
}

rail_depot = {
	name = "Rail Depot"

	}
}
